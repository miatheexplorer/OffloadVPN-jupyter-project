=====Flow policy verification=====
hgw sh -c 'tcpdump -nn -i eth0 "tcp[tcpflags] & tcp-syn != 0 and dst port 5001" -c 2 > /tmp/hgw5001.txt 2>&1 &'
h2 sh -c 'timeout 5 nc -l 5001 > /tmp/recv5001.txt &'
sh sleep 1
h1 sh -c 'echo hello | timeout 3 nc 10.0.2.2 5001'
sh sleep 1

hgw cat /tmp/hgw5001.txt
h2 cat /tmp/recv5001.txt

=====Demonstrate offloading behavior=====
hgw sh -c 'tcpdump -nn -i eth0 "tcp[tcpflags] & tcp-syn != 0 and (dst port 5000 or dst port 5001)" -c 4 > /tmp/hgw_flows.txt 2>&1 &'

h2 sh -c 'timeout 5 nc -l 5000 > /tmp/recv5000.txt &'
sh sleep 1
h1 sh -c 'echo first | timeout 3 nc 10.0.2.2 5000'
sh sleep 1

h2 sh -c 'timeout 5 nc -l 5001 > /tmp/recv5001.txt &'
sh sleep 1
h1 sh -c 'echo second | timeout 3 nc 10.0.2.2 5001'
sh sleep 1

hgw cat /tmp/hgw_flows.txt
h2 cat /tmp/recv5000.txt
h2 cat /tmp/recv5001.txt
